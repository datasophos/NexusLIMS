1""" 

2Extract metadata from various electron microscopy file types. 

3 

4Extractors should return a list of dictionaries, where each dictionary contains 

5the extracted metadata under the key ``nx_meta``. The ``nx_meta`` structure is 

6validated against the :class:`~nexusLIMS.extractors.schemas.NexusMetadata` Pydantic 

7schema to ensure consistency across all extractors. 

8 

9Required Fields 

10--------------- 

11All extractors must include these fields in ``nx_meta``: 

12 

13* ``'Creation Time'`` - ISO-8601 timestamp string **with timezone** (e.g., 

14 ``"2024-01-15T10:30:00-05:00"`` or ``"2024-01-15T15:30:00Z"``) 

15* ``'Data Type'`` - Human-readable description using underscores (e.g., 

16 ``"STEM_Imaging"``, ``"TEM_EDS"``, ``"SEM_Imaging"``) 

17* ``'DatasetType'`` - Schema-defined category, must be one of: ``"Image"``, 

18 ``"Spectrum"``, ``"SpectrumImage"``, ``"Diffraction"``, ``"Misc"``, or ``"Unknown"`` 

19 

20Optional Fields 

21--------------- 

22Common optional fields include: 

23 

24* ``'Data Dimensions'`` - Dataset shape as string (e.g., ``"(1024, 1024)"``) 

25* ``'Instrument ID'`` - Instrument PID from database (e.g., ``"FEI-Titan-TEM-635816"``) 

26* ``'warnings'`` - List of warning messages or [message, context] pairs 

27 

28Additional instrument-specific fields are allowed beyond these standard fields. 

29 

30Schema Validation 

31----------------- 

32The ``nx_meta`` structure is validated using Pydantic strict mode. Validation occurs 

33after default values are set (e.g., missing ``DatasetType`` defaults to ``"Misc"``). 

34If validation fails, a ``pydantic.ValidationError`` is raised with detailed information 

35about which fields are invalid. 

36 

37For complete schema details, see :class:`~nexusLIMS.extractors.schemas.NexusMetadata`. 

38"""

102def _add_extraction_details( 

103 nx_meta: Dict, 

104 extractor_module: Callable, 

105) -> Dict[str, str]: 

106 """ 

107 Add extraction details to the NexusLIMS metadata. 

108 

109 Adds metadata about the extraction process, given an extractor module 

110 to the ``nx_meta`` metadata dictionary under the ``'NexusLIMS Extraction'`` 

111 sub-key. The ``'Extractor Module'`` metadata key will contain the fully 

112 qualified path of a given extractor, e.g. 

113 ``nexusLIMS.extractors.basic_metadata``. 

114 

115 Note 

116 ---- 

117 If the ``'NexusLIMS Extraction'`` key already exists in the ``nx_meta`` 

118 metadata dictionary, this method *will* overwrite its value. 

119 

120 Parameters 

121 ---------- 

122 nx_meta 

123 The metadata dictionary as returend by :py:meth:`parse_metadata` 

124 extractor_module 

125 The (callable) module for a specific metadata extractor from the 

126 :py:mod:`~nexusLIMS.extractors` module. 

127 

128 Returns 

129 ------- 

130 dict 

131 An updated ``nx_meta`` dictionary, containing extraction details 

132 

133 """ 

134 # PHASE 1 MIGRATION: Handle both old-style functions and new-style extractors 

135 # Try to get the module name in different ways for backward compatibility 

136 module_name = None 

137 

138 # Try __module__ attribute first (works for new extractor system) 

139 if hasattr(extractor_module, "__module__"): 

140 module_name = extractor_module.__module__ 

141 

142 # Fallback to inspect.getmodule() for old-style functions 

143 if module_name is None: # pragma: no cover 

144 module = inspect.getmodule(extractor_module) # pragma: no cover 

145 # Last resort - use "unknown" 

146 module_name = ( # pragma: no cover 

147 module.__name__ if module is not None else "unknown" 

148 ) 

149 

150 nx_meta["nx_meta"]["NexusLIMS Extraction"] = { 

151 "Date": dt.now(tz=current_system_tz()).isoformat(), 

152 "Module": module_name, 

153 "Version": __version__, 

154 } 

155 

156 return nx_meta

159def validate_nx_meta( 

160 metadata_dict: dict[str, Any], *, filename: Path | None = None 

161) -> dict[str, Any]: 

162 """ 

163 Validate the nx_meta structure against the NexusMetadata schema. 

164 

165 This function ensures that metadata returned by extractor plugins conforms 

166 to the required structure defined in 

167 :class:`~nexusLIMS.extractors.schemas.NexusMetadata`. Validation is 

168 performed strictly - any schema violations will raise a ValidationError 

169 with detailed information about the failure. 

170 

171 Parameters 

172 ---------- 

173 metadata_dict : dict[str, Any] 

174 Dictionary containing an 'nx_meta' key with the metadata to validate. 

175 This is the format returned by all extractor plugins. 

176 filename : Path or None, optional 

177 The file path being processed. Used only for error message context. 

178 If None, error messages will not include file path information. 

179 

180 Returns 

181 ------- 

182 dict[str, Any] 

183 The original metadata_dict, unchanged. Validation does not modify data, 

184 it only checks conformance to the schema. 

185 

186 Raises 

187 ------ 

188 ValidationError 

189 If the nx_meta structure fails validation. The error message will include 

190 detailed information about which fields are invalid and why. 

191 

192 Notes 

193 ----- 

194 This function validates: 

195 

196 - **Required fields**: 'Creation Time', 'Data Type', 'DatasetType' must be present 

197 - **ISO-8601 timestamps**: 'Creation Time' must be valid ISO-8601 with timezone 

198 - **Controlled vocabularies**: 'DatasetType' must be one of the allowed values 

199 - **Type constraints**: All fields must match their expected types 

200 

201 Additional fields beyond the schema are allowed (for instrument-specific metadata). 

202 

203 Examples 

204 -------- 

205 Valid metadata passes without modification: 

206 

207 >>> metadata = { 

208 ... "nx_meta": { 

209 ... "Creation Time": "2024-01-15T10:30:00-05:00", 

210 ... "Data Type": "STEM_Imaging", 

211 ... "DatasetType": "Image", 

212 ... } 

213 ... } 

214 >>> result = validate_nx_meta(metadata) 

215 >>> result == metadata 

216 True 

217 

218 Invalid metadata raises ValidationError: 

219 

220 >>> bad_metadata = { 

221 ... "nx_meta": { 

222 ... "Creation Time": "invalid-timestamp", 

223 ... "Data Type": "STEM_Imaging", 

224 ... "DatasetType": "Image", 

225 ... } 

226 ... } 

227 >>> validate_nx_meta(bad_metadata) # doctest: +SKIP 

228 Traceback (most recent call last): 

229 ... 

230 pydantic.ValidationError: ... 

231 

232 See Also 

233 -------- 

234 NexusMetadata : The Pydantic schema model for nx_meta validation 

235 parse_metadata : Main extraction function that uses this validator 

236 """ 

237 try: 

238 NexusMetadata.model_validate(metadata_dict["nx_meta"]) 

239 except ValidationError as e: 

240 # Enhance error message with file context if available 

241 if filename: 

242 msg = f"Validation failed for {filename}: {e}" 

243 else: 

244 msg = f"Validation failed: {e}" 

245 logger.exception(msg) 

246 raise 

247 

248 return metadata_dict

251def parse_metadata( # noqa: PLR0912 

252 fname: Path, 

253 *, 

254 write_output: bool = True, 

255 generate_preview: bool = True, 

256 overwrite: bool = True, 

257) -> Tuple[Dict[str, Any] | None, Path | list[Path] | None]: 

258 """ 

259 Parse metadata from a file and optionaly generate a preview image. 

260 

261 Given an input filename, read the file, determine what "type" of file (i.e. 

262 what instrument it came from) it is, filter the metadata (if necessary) to 

263 what we are interested in, and return it as a dictionary (writing to the 

264 NexusLIMS directory as JSON by default). Also calls the preview 

265 generation method, if desired. 

266 

267 For files containing multiple signals (e.g., multi-signal DM3/DM4 files), 

268 generates one preview per signal and returns a list of preview paths. 

269 

270 Parameters 

271 ---------- 

272 fname 

273 The filename from which to read data 

274 write_output 

275 Whether to write the metadata dictionary as a json file in the NexusLIMS 

276 folder structure 

277 generate_preview 

278 Whether to generate the thumbnail preview of this dataset (that 

279 operation is not done in this method, it is just called from here so 

280 it can be done at the same time) 

281 overwrite 

282 Whether to overwrite the .json metadata file and thumbnail 

283 image if either exists 

284 

285 Returns 

286 ------- 

287 nx_meta : list[dict] or None 

288 A list of metadata dicts, one per signal in the file. If None, 

289 the file could not be opened. Single-signal files return a list 

290 with one dict, multi-signal files return a list with multiple dicts. 

291 preview_fname : list[Path] or None 

292 A list of file paths for the generated preview images, one per signal. 

293 For single-signal files, returns a list with one path. Returns `None` 

294 if preview generation was not requested. 

295 """ 

296 extension = fname.suffix[1:] 

297 

298 # Create extraction context 

299 instrument = get_instr_from_filepath(fname) 

300 context = ExtractionContext(file_path=fname, instrument=instrument) 

301 

302 # Get extractor from registry 

303 registry = get_registry() 

304 extractor = registry.get_extractor(context) 

305 

306 # Extract metadata using the selected extractor 

307 # All extractors now return a list of dicts (one per signal) 

308 nx_meta_list = extractor.extract(context) 

309 

310 # Create a pseudo-module for extraction details tracking 

311 class ExtractorMethod: 

312 """Pseudo-module for extraction details tracking.""" 

313 

314 def __init__(self, extractor_name: str): 

315 # Use the plugin module path for all extractors 

316 self.__module__ = f"nexusLIMS.extractors.plugins.{extractor_name}" 

317 self.__name__ = self.__module__ 

318 

319 def __call__(self, f: Path) -> dict: # noqa: ARG002 

320 return nx_meta_list # pragma: no cover 

321 

322 # Defensive check: extractors should always return a list but handle None gracefully 

323 if nx_meta_list is None: 

324 return None, None 

325 

326 extractor_method = ExtractorMethod(extractor.name) 

327 

328 # Handle preview generation logic if the extractor is 

329 # the basic fallback and extension is not in unextracted_preview_map, 

330 # don't generate a preview 

331 if extractor.name == "basic_file_info_extractor": 

332 if extension not in unextracted_preview_map: 

333 generate_preview = False 

334 logger.info( 

335 "No specialized extractor found for file extension; " 

336 "setting generate_preview to False", 

337 ) 

338 else: 

339 generate_preview = True 

340 logger.info( 

341 "No specialized extractor found for file extension; " 

342 "but file extension was in unextracted_preview_map; " 

343 "setting generate_preview to True", 

344 ) 

345 

346 # Add extraction details to metadata 

347 nx_meta_list = [_add_extraction_details(m, extractor_method) for m in nx_meta_list] 

348 

349 signal_count = len(nx_meta_list) 

350 preview_fnames = [] 

351 

352 # Set the dataset type to Misc if it was not set by the file reader 

353 for nx_meta in nx_meta_list: 

354 if "DatasetType" not in nx_meta["nx_meta"]: 

355 nx_meta["nx_meta"]["DatasetType"] = "Misc" 

356 nx_meta["nx_meta"]["Data Type"] = "Miscellaneous" 

357 

358 # Validate each metadata dict against the schema (strict mode) 

359 # This happens AFTER setting defaults to allow extractors to omit optional fields 

360 for nx_meta in nx_meta_list: 

361 validate_nx_meta(nx_meta, filename=fname) 

362 

363 # Write output for each signal (single and multi-signal files) 

364 if write_output: 

365 for i, nx_meta in enumerate(nx_meta_list): 

366 # For single-signal files, omit suffix for backward compatibility 

367 if signal_count == 1: 

368 out_fname = replace_instrument_data_path(fname, ".json") 

369 else: 

370 # For multi-signal files, append signal index to filename 

371 base_path = replace_instrument_data_path(fname, "") 

372 out_fname = Path(f"{base_path}_signal{i}.json") 

373 

374 if not out_fname.exists() or overwrite: 

375 # Create the directory for the metadata file, if needed 

376 out_fname.parent.mkdir(parents=True, exist_ok=True) 

377 # Make sure that the nx_meta dict comes first in the json output 

378 out_dict = {"nx_meta": nx_meta["nx_meta"]} 

379 for k, v in nx_meta.items(): 

380 if k == "nx_meta": 

381 pass 

382 else: 

383 out_dict[k] = v 

384 with out_fname.open(mode="w", encoding="utf-8") as f: 

385 logger.debug("Dumping metadata to %s", out_fname) 

386 json.dump( 

387 out_dict, 

388 f, 

389 sort_keys=False, 

390 indent=2, 

391 cls=_CustomEncoder, 

392 ) 

393 

394 # Generate previews for each signal 

395 if generate_preview: 

396 for i in range(signal_count): 

397 # For single-signal files, omit suffix for backward compatibility 

398 signal_idx = i if signal_count > 1 else None 

399 preview = create_preview( 

400 fname=fname, 

401 overwrite=overwrite, 

402 signal_index=signal_idx, 

403 ) 

404 preview_fnames.append(preview) 

405 else: 

406 preview_fnames = [None] * signal_count 

407 

408 return nx_meta_list, preview_fnames

411def create_preview( # noqa: PLR0911, PLR0912, PLR0915 

412 fname: Path, *, overwrite: bool, signal_index: int | None = None 

413) -> Path | None: 

414 """ 

415 Generate a preview image for a given file using the plugin system. 

416 

417 This method uses the preview generator plugin system to create thumbnail 

418 previews. It first tries to find a suitable preview generator plugin, and 

419 falls back to legacy methods if no plugin is found. 

420 

421 Parameters 

422 ---------- 

423 fname 

424 The filename from which to read data 

425 overwrite 

426 Whether to overwrite the .json metadata file and thumbnail 

427 image if either exists 

428 signal_index 

429 For files with multiple signals, the index of the signal to preview. 

430 If None, generates a single preview (legacy behavior). If an int, 

431 generates preview with _signalN suffix in filename. 

432 

433 Returns 

434 ------- 

435 preview_fname : Optional[pathlib.Path] 

436 The filename of the generated preview image; if None, a preview could not be 

437 successfully generated. 

438 """ 

439 # Generate preview filename with signal index suffix if provided 

440 if signal_index is None: 

441 preview_fname = replace_instrument_data_path(fname, ".thumb.png") 

442 else: 

443 preview_fname = replace_instrument_data_path( 

444 fname, f"_signal{signal_index}.thumb.png" 

445 ) 

446 

447 # Skip if preview exists and overwrite is False 

448 if preview_fname.is_file() and not overwrite: 

449 logger.info("Preview already exists: %s", preview_fname) 

450 return preview_fname 

451 

452 # Create context for preview generation 

453 instrument = get_instr_from_filepath(fname) 

454 context = ExtractionContext( 

455 file_path=fname, instrument=instrument, signal_index=signal_index 

456 ) 

457 

458 # Try to get a preview generator from the registry 

459 registry = get_registry() 

460 generator = registry.get_preview_generator(context) 

461 

462 if generator: 

463 # Use plugin-based preview generation 

464 logger.info("Generating preview using %s: %s", generator.name, preview_fname) 

465 # Create the directory for the thumbnail, if needed 

466 preview_fname.parent.mkdir(parents=True, exist_ok=True) 

467 

468 success = generator.generate(context, preview_fname) 

469 if success: 

470 return preview_fname 

471 

472 logger.warning( 

473 "Preview generator %s failed for %s", 

474 generator.name, 

475 fname, 

476 ) 

477 # Fall through to legacy methods 

478 

479 # Legacy fallback for .tif files (special case with downsampling) 

480 extension = fname.suffix[1:] 

481 if extension == "tif": 

482 logger.info("Using legacy downsampling for .tif: %s", preview_fname) 

483 preview_fname.parent.mkdir(parents=True, exist_ok=True) 

484 factor = 2 

485 down_sample_image(fname, out_path=preview_fname, factor=factor) 

486 return preview_fname 

487 

488 # Legacy fallback for files in unextracted_preview_map 

489 if extension in unextracted_preview_map: 

490 logger.info("Using legacy preview map for %s: %s", extension, preview_fname) 

491 preview_fname.parent.mkdir(parents=True, exist_ok=True) 

492 preview_return = unextracted_preview_map[extension]( 

493 f=fname, 

494 out_path=preview_fname, 

495 output_size=500, 

496 ) 

497 

498 # handle the case where PIL cannot open an image 

499 if preview_return is False: 

500 return None 

501 

502 return preview_fname 

503 

504 # Legacy fallback for HyperSpy-loadable files 

505 logger.info("Trying legacy HyperSpy preview generation: %s", preview_fname) 

506 load_options = {"lazy": True} 

507 if extension == "ser": 

508 load_options["only_valid_data"] = True 

509 

510 # noinspection PyBroadException 

511 try: 

512 s = hs.load(fname, **load_options) 

513 except Exception: # pylint: disable=broad-exception-caught 

514 logger.warning( 

515 "Signal could not be loaded by HyperSpy. " 

516 "Using placeholder image for preview.", 

517 ) 

518 preview_fname.parent.mkdir(parents=True, exist_ok=True) 

519 shutil.copyfile(PLACEHOLDER_PREVIEW, preview_fname) 

520 return preview_fname 

521 

522 # If s is a list of signals, select the appropriate one 

523 if isinstance(s, list): 

524 num_sigs = len(s) 

525 original_fname = s[0].metadata.General.original_filename 

526 if signal_index is not None: 

527 # Use specified signal index 

528 s = s[signal_index] 

529 s.metadata.General.title = ( 

530 s.metadata.General.title 

531 + f" (signal {signal_index + 1} of " 

532 + f'{num_sigs} in file "{original_fname}")' 

533 ) 

534 else: 

535 # Legacy: use first signal only 

536 s = s[0] 

537 s.metadata.General.title = ( 

538 s.metadata.General.title 

539 + f' (1 of {num_sigs} total signals in file "{original_fname}")' 

540 ) 

541 elif not s.metadata.General.title: 

542 s.metadata.General.title = s.metadata.General.original_filename.replace( 

543 extension, 

544 "", 

545 ).strip(".") 

546 

547 # Generate the preview 

548 logger.info("Generating HyperSpy preview: %s", preview_fname) 

549 preview_fname.parent.mkdir(parents=True, exist_ok=True) 

550 s.compute(show_progressbar=False) 

551 sig_to_thumbnail(s, out_path=preview_fname) 

552 

553 return preview_fname

556def flatten_dict(_dict, parent_key="", separator=" "): # noqa: ARG001 

557 """ 

558 Flatten a nested dictionary into a single level. 

559 

560 Utility method to take a nested dictionary structure and flatten it into a 

561 single level, separating the levels by a string as specified by 

562 ``separator``. 

563 

564 Uses python-benedict for robust nested dictionary operations. 

565 

566 Parameters 

567 ---------- 

568 _dict : dict 

569 The dictionary to flatten 

570 parent_key : str 

571 The "root" key to add to the existing keys (unused in current implementation) 

572 separator : str 

573 The string to use to separate values in the flattened keys (i.e. 

574 {'a': {'b': 'c'}} would become {'a' + sep + 'b': 'c'}) 

575 

576 Returns 

577 ------- 

578 flattened_dict : str 

579 The dictionary with depth one, with nested dictionaries flattened 

580 into root-level keys 

581 """ 

582 # Disable keypath_separator to avoid conflicts with keys containing 

583 # dots or other special chars 

584 return benedict(_dict, keypath_separator=None).flatten(separator=separator)

587class _CustomEncoder(json.JSONEncoder): 

588 """ 

589 Allow non-serializable types to be written in a JSON format. 

590 

591 A custom JSON Encoder class that will allow certain types to be serialized that are 

592 not able to be by default (taken from https://stackoverflow.com/a/27050186). 

593 """ 

594 

595 def default(self, o): 

596 if isinstance(o, np.integer): 

597 return int(o) 

598 if isinstance(o, np.floating): 

599 return float(o) 

600 if isinstance(o, np.ndarray): 

601 return o.tolist() 

602 if isinstance(o, np.bytes_): 

603 return o.decode() 

604 if isinstance(o, np.void): 

605 # np.void array may contain arbitary binary, so base64 encode it 

606 return base64.b64encode(o.tolist()).decode("utf-8") 

607 

608 return super().default(o)
