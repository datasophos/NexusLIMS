1""" 

2NEMO harvester module. 

3 

4This module contains the functionality to harvest instruments, reservations, 

5etc. from an instance of NEMO (https://github.com/usnistgov/NEMO/), a 

6calendering and laboratory logistics application. 

7"""

10from datetime import timedelta 

11 

12from nexusLIMS.db.session_handler import Session

17from .utils import ( 

18 _get_res_question_value, 

19 get_connector_for_session, 

20 id_from_url, 

21 process_res_question_samples, 

22)

24logger = logging.getLogger(__name__)

27def res_event_from_session(session: Session) -> ReservationEvent: 

28 """ 

29 Create reservation event from session. 

30 

31 Create an internal 

32 :py:class:`~nexusLIMS.harvesters.reservation_event.ReservationEvent` representation 

33 of a session by finding a matching reservation in the NEMO 

34 system and parsing the data contained within into a ``ReservationEvent``. 

35 

36 This method assumes a certain format for the "reservation questions" 

37 associated with each reservation and parses that information into the resulting 

38 ``ReservationEvent``. The most critical of these is the ``data_consent`` field. 

39 If an affirmative response in this field is not found (because the user declined 

40 consent or the reservation questions are missing), a record will not be built. 

41 

42 The following JSON object represents a minimal schema for a set of NEMO "Reservation 

43 Questions" that will satisfy the expectations of this method. Please see the 

44 NEMO documentation on this feature for more details. 

45 

46 .. highlight:: json 

47 .. code-block:: json 

48 

49 [ 

50 { 

51 "type": "textbox", 

52 "name": "project_id", 

53 "title": "Project ID", 

54 }, 

55 { 

56 "type": "textbox", 

57 "name": "experiment_title", 

58 "title": "Title of Experiment", 

59 }, 

60 { 

61 "type": "textarea", 

62 "name": "experiment_purpose", 

63 "title": "Experiment Purpose", 

64 }, 

65 { 

66 "type": "radio", 

67 "title": "Agree to NexusLIMS curation", 

68 "choices": ["Agree", "Disagree"], 

69 "name": "data_consent", 

70 "default_choice": "Agree" 

71 }, 

72 { 

73 "type": "group", 

74 "title": "Sample information", 

75 "name": "sample_group", 

76 "questions": [ 

77 { 

78 "type": "textbox", 

79 "name": "sample_name", 

80 "title": "Sample Name / PID", 

81 }, 

82 { 

83 "type": "radio", 

84 "title": "Sample or PID?", 

85 "choices": ["Sample Name", "PID"], 

86 "name": "sample_or_pid", 

87 }, 

88 { 

89 "type": "textarea", 

90 "name": "sample_details", 

91 "title": "Sample Details", 

92 } 

93 ] 

94 } 

95 ] 

96 

97 Parameters 

98 ---------- 

99 session 

100 The session for which to get a reservation event 

101 

102 Returns 

103 ------- 

104 res_event : ~nexusLIMS.harvesters.reservation_event.ReservationEvent 

105 The matching reservation event 

106 """ 

107 # a session has instrument, dt_from, dt_to, and user 

108 

109 # we should fetch all reservations +/- two days, and then find the one 

110 # with the maximal overlap with the session time range 

111 # probably don't want to filter by user for now, since sometimes users 

112 # will enable/reserve on behalf of others, etc. 

113 

114 # in order to get reservations, we need a NemoConnector 

115 nemo_connector = get_connector_for_session(session) 

116 

117 # get reservation with maximum overlap 

118 reservations = nemo_connector.get_reservations( 

119 # tool id can be extracted from instrument api_url query parameter 

120 tool_id=id_from_url(session.instrument.api_url), 

121 dt_from=session.dt_from - timedelta(days=2), 

122 dt_to=session.dt_to + timedelta(days=2), 

123 ) 

124 

125 logger.info( 

126 "Found %i reservations between %s and %s with ids: %s", 

127 len(reservations), 

128 session.dt_from - timedelta(days=2), 

129 session.dt_to + timedelta(days=2), 

130 [i["id"] for i in reservations], 

131 ) 

132 for i, res in enumerate(reservations): 

133 logger.debug( 

134 "Reservation %i: %sreservations/?id=%s from %s to %s", 

135 i + 1, 

136 nemo_connector.config["base_url"], 

137 res["id"], 

138 res["start"], 

139 res["end"], 

140 ) 

141 

142 starts = [nemo_connector.strptime(r["start"]) for r in reservations] 

143 ends = [nemo_connector.strptime(r["end"]) for r in reservations] 

144 

145 overlaps = [ 

146 get_timespan_overlap((session.dt_from, session.dt_to), (s, e)) 

147 for s, e in zip(starts, ends) 

148 ] 

149 

150 # handle if there are no matching sessions (i.e. reservations is an empty list 

151 # also need to handle if there is no overlap at all with any reservation 

152 if len(reservations) == 0 or max(overlaps) == timedelta(0): 

153 # there were no reservations that matched this usage event time range, 

154 # or none of the reservations overlapped with the usage event 

155 # so we'll use what limited information we have from the usage event 

156 # session 

157 logger.warning( 

158 "No reservations found with overlap for this usage " 

159 "event, so raising NoDataConsentError", 

160 ) 

161 msg = ( 

162 "No reservation found matching this session, so assuming NexusLIMS " 

163 "does not have user consent for data harvesting." 

164 ) 

165 raise NoMatchingReservationError(msg) 

166 

167 # select the reservation with the most overlap 

168 res = reservations[overlaps.index(max(overlaps))] 

169 logger.info( 

170 "Using reservation %sreservations/?id=%s as match for " 

171 "usage event %s with overlap of %s", 

172 nemo_connector.config["base_url"], 

173 res["id"], 

174 session.session_identifier, 

175 max(overlaps), 

176 ) 

177 

178 # DONE: check for presence of sample_group in the reservation metadata 

179 # and change the harvester to process the sample group metadata by 

180 # providing lists to the ReservationEvent constructor 

181 ( 

182 sample_details, 

183 sample_pid, 

184 sample_name, 

185 sample_elements, 

186 ) = process_res_question_samples(res) 

187 

188 # DONE: respect user choice not to harvest data (data_consent) 

189 consent = "disagree" 

190 consent = _get_res_question_value("data_consent", res) 

191 # consent will be None here if it wasn't given (i.e. there was no 

192 # data_consent field in the reservation questions) 

193 if consent is None: 

194 msg = ( 

195 f"Reservation {res['id']} did not have data_consent defined, " 

196 "so we should not harvest its data" 

197 ) 

198 raise NoDataConsentError(msg) 

199 

200 if consent.lower() in ["disagree", "no", "false", "negative"]: 

201 msg = f"Reservation {res['id']} requested not to have their data harvested" 

202 raise NoDataConsentError(msg) 

203 

204 # Create ReservationEvent from NEMO reservation dict 

205 return ReservationEvent( 

206 experiment_title=_get_res_question_value("experiment_title", res), 

207 instrument=session.instrument, 

208 last_updated=nemo_connector.strptime(res["creation_time"]), 

209 username=res["user"]["username"], 

210 user_full_name=( 

211 f"{res['user']['first_name']} " 

212 f"{res['user']['last_name']} " 

213 f"({res['user']['username']})" 

214 ), 

215 created_by=res["creator"]["username"], 

216 created_by_full_name=( 

217 f"{res['creator']['first_name']} " 

218 f"{res['creator']['last_name']} " 

219 f"({res['creator']['username']})" 

220 ), 

221 start_time=nemo_connector.strptime(res["start"]), 

222 end_time=nemo_connector.strptime(res["end"]), 

223 reservation_type=None, # reservation type is not collected in NEMO 

224 experiment_purpose=_get_res_question_value("experiment_purpose", res), 

225 sample_details=sample_details, 

226 sample_pid=sample_pid, 

227 sample_name=sample_name, 

228 sample_elements=sample_elements, 

229 project_name=[None], 

230 project_id=[_get_res_question_value("project_id", res)], 

231 project_ref=[None], 

232 internal_id=str(res["id"]), 

233 division=None, 

234 group=None, 

235 url=nemo_connector.config["base_url"].replace( 

236 "api/", 

237 f"event_details/reservation/{res['id']}/", 

238 ), 

239 )
